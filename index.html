      const holidays = [{ name: "春節", date: "2026-01-28" }, { name: "和平紀念日", date: "2026-02-28" }, { name: "清明節", date: "2026-04-04" }, { name: "端午節", date: "2026-06-20" }, { name: "中秋節", date: "2026-10-06" }, { name: "國慶日", date: "2026-10-10" }];
      const now = new Date();
      let next = holidays.find(h => new Date(h.date) > now) || { name: "新年", date: "2027-01-01" };
      const days = Math.ceil((new Date(next.date) - now) / (1000 * 60 * 60 * 24));
      document.getElementById('holiday-countdown').innerText = `📅 ${next.name}${days}天`;
      }
      function updateMarketStatus() {
      const now = new Date();
      document.getElementById('clock-time').innerText = now.toLocaleTimeString('zh-TW', {hour12: false});
      const twTimer = document.getElementById('market-timer');
      const day = now.getDay(), h = now.getHours(), m = now.getMinutes();
      if (day === 0 || day === 6) twTimer.innerText = "休市";
      else if (h < 9) twTimer.innerText = "未開盤";
      else if (h >= 13 && m >= 30 || h > 13) twTimer.innerText = "已收盤";
      else twTimer.innerHTML = `<span class="red">開盤中</span>`;
      }

      function unifiedSearch(val) {

      function fetchPrice(code) {
      const url = `https://corsproxy.io/?${encodeURIComponent(`https://query1.finance.yahoo.com/v8/finance/chart/${code}.TW?interval=1d`)}`;
      fetch(url).then(r => r.json()).then(d => {
      if (d.chart.result) {
      let m = d.chart.result[0].meta, p = m.regularMarketPrice, prev = m.chartPreviousClose, chg = p - prev, pct = (chg/prev)*100;
      const clr = chg > 0 ? "red" : (chg < 0 ? "green" : "");
      document.getElementById("prev-"+code).innerText = prev.toFixed(1);
      document.getElementById("p-"+code).innerText = p.toFixed(1); document.getElementById("p-"+code).className = clr;
      document.getElementById("c-"+code).innerText = (chg > 0 ? "▲" : "▼") + Math.abs(chg).toFixed(2);
      document.getElementById("c-"+code).className = clr;
      document.getElementById("pct-"+code).innerText = pct.toFixed(2) + "%"; document.getElementById("pct-"+code).className = clr;
      }
      });
      }
